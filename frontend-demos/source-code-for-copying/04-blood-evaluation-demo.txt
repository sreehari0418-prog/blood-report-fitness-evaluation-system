    <script>
        let selectedFile = null;

        document.getElementById('fileInput').addEventListener('change', (e) => {
            const file = e.target.files[0];
            if (file) {
                selectedFile = file;
                const preview = document.getElementById('filePreview');
                preview.innerHTML = `ðŸ“Ž Selected: <strong>${file.name}</strong> (${(file.size / 1024).toFixed(1)} KB)`;
                preview.style.display = 'block';

                document.getElementById('analyzeBtn').classList.add('show');
                console.log('File selected:', file.name, file.type, file.size);
            }
        });

        document.getElementById('analyzeBtn').addEventListener('click', () => {
            if (!selectedFile) {
                alert('Please select a file first!');
                return;
            }

            // Hide upload card
            document.querySelector('.upload-card').style.display = 'none';

            // Show processing
            document.getElementById('processing').style.display = 'block';

            // Simulate processing
            setTimeout(() => {
                document.getElementById('processing').style.display = 'none';
                document.getElementById('analysisCard').classList.add('show');

                console.log('Analysis complete!');
                console.log('Sample blood parameters extracted and analyzed using ML model');
            }, 2500);
        });
    </script>
